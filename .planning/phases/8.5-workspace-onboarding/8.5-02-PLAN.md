4. Create `packages/cli/src/daemon.ts` implementing the lazy daemon + proxy pattern (per D-02, D-07):
   ```typescript
   import { spawn } from 'node:child_process';
   import { resolve } from 'node:path';

   const DAEMON_PORT = parseInt(process.env.BRIDGE_DAEMON_PORT ?? '3000', 10);
   const DAEMON_HOST = process.env.BRIDGE_DAEMON_HOST ?? '127.0.0.1';
   const HEALTH_URL = `http://${DAEMON_HOST}:${DAEMON_PORT}/health`;

   export interface DaemonStatus {
     running: boolean;
     url: string;
   }

   /** Check if daemon is alive by pinging /health. */
   export async function checkDaemon(): Promise<DaemonStatus> {
     const url = `http://${DAEMON_HOST}:${DAEMON_PORT}`;
     try {
       const res = await fetch(HEALTH_URL, { signal: AbortSignal.timeout(3000) });
       if (res.ok) {
         return { running: true, url };
       }
       return { running: false, url };
     } catch {
       return { running: false, url };
     }
   }

   /** Start the daemon as a detached background process. */
   export async function startDaemon(): Promise<void> {
     // Resolve the API entry point relative to this package
     // In dev: the api app is at apps/api/
     const apiEntry = resolve(process.cwd(), 'apps', 'api', 'src', 'main.ts');

     console.log('[bridge] Starting daemon...');
     const child = spawn('bun', ['run', apiEntry], {
       detached: true,
       stdio: 'ignore',
       env: {
         ...process.env,
         PORT: String(DAEMON_PORT),
       },
     });
     child.unref();

     // Wait for daemon to become healthy (up to 15 seconds)
     const deadline = Date.now() + 15_000;
     while (Date.now() < deadline) {
       await new Promise(r => setTimeout(r, 500));
       const status = await checkDaemon();
       if (status.running) {
         console.log('[bridge] Daemon started successfully');
         return;
       }
     }
     throw new Error('Daemon failed to start within 15 seconds');
   }

   /** Ensure daemon is running — start if needed. Returns the base URL. */
   export async function ensureDaemon(): Promise<string> {
     const status = await checkDaemon();
     if (status.running) {
       return status.url;
     }
     await startDaemon();
     return `http://${DAEMON_HOST}:${DAEMON_PORT}`;
   }
   ```

5. Create `packages/cli/src/commands/init.ts`:
   ```typescript
   import { resolve } from 'node:path';
   import { existsSync } from 'node:fs';
   import { ensureDaemon } from '../daemon.js';

   export interface InitOptions {
     workspace: string;
     repo?: string;
     name?: string;
     slug?: string;
   }

   /**
    * Execute `bridge init` — the primary onboarding command (per D-01, D-08).
    *
    * Two sequential stages:
    * 1. Offline extraction (git, manifests, tree, entrypoints, docs)
    * 2. GSD InitRunner brownfield mode (AI doc generation)
    *
    * For now, this sends the command to the daemon which orchestrates both stages.
    */
   export async function runInit(options: InitOptions): Promise<void> {
     const workspacePath = resolve(options.workspace);

     // Validate workspace path exists
     if (!existsSync(workspacePath)) {
       console.error(`[bridge] Error: workspace path does not exist: ${workspacePath}`);
       process.exit(1);
     }

     console.log(`[bridge] Initializing workspace: ${workspacePath}`);
     if (options.repo) {
       console.log(`[bridge] Repo URL: ${options.repo}`);
     }

     // Ensure daemon is running (lazy start per D-02)
     let daemonUrl: string;
     try {
       daemonUrl = await ensureDaemon();
     } catch (err) {
       console.error(`[bridge] Failed to connect to daemon: ${err instanceof Error ? err.message : err}`);
       console.error('[bridge] Make sure the bridge-ai API is available or start it manually.');
       process.exit(1);
     }

     console.log(`[bridge] Connected to daemon at ${daemonUrl}`);

     // Derive project name and slug from workspace path if not provided
     const projectName = options.name ?? workspacePath.split('/').pop() ?? 'unnamed';
     const projectSlug = options.slug ?? projectName.toLowerCase().replace(/[^a-z0-9-]/g, '-').replace(/-+/g, '-');

     // TODO: Plan 06 wires this to the actual daemon endpoint.
     // For now, print what would be sent.
     console.log(`[bridge] Would onboard project: ${projectName} (${projectSlug})`);
     console.log(`[bridge] Workspace: ${workspacePath}`);
     console.log('[bridge] Init command registered. Wiring to daemon happens in Plan 06.');
   }
   ```
