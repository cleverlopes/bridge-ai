```typescript
@Injectable()
export class RepoIndexerService {
  private readonly logger = new Logger(RepoIndexerService.name);

  private static readonly MAX_TREE_ENTRIES = 2000;
  private static readonly MAX_DEPTH = 3;

  private static readonly MANIFEST_PATTERNS: Record<string, string> = {
    'package.json': 'npm',
    'pyproject.toml': 'python',
    'Cargo.toml': 'rust',
    'go.mod': 'go',
    'pom.xml': 'maven',
    'build.gradle': 'gradle',
    'build.gradle.kts': 'gradle',
  };

  private static readonly ENTRYPOINT_PATTERNS = [
    'main.ts', 'index.ts', 'app.ts', 'main.js', 'index.js', 'app.js',
    'main.py', '__main__.py', 'app.py',
    'main.go', 'cmd/main.go',
    'src/main.ts', 'src/index.ts', 'src/app.ts',
    'src/main.rs', 'src/lib.rs',
  ];

  private static readonly TEST_PATTERNS = /\.(spec|test)\.(ts|js|tsx|jsx)$|^test\/|^tests\/|^__tests__\//;

  private static readonly DOC_FILES = [
    'README.md', 'CHANGELOG.md', 'CONTRIBUTING.md', 'LICENSE', 'LICENSE.md',
  ];
  private static readonly DOC_DIRS = ['docs', '.planning'];

  async bootstrap(workspacePath: string, repoInfo: RepoInfo): Promise<IndexPayload> { ... }
  private async walkTree(basePath: string, relativePath: string, depth: number, results: string[]): Promise<boolean> { ... }
  private async detectManifests(workspacePath: string, tree: string[]): Promise<ManifestEntry[]> { ... }
  private detectEntrypoints(tree: string[]): string[] { ... }
  private detectTestPaths(tree: string[]): string[] { ... }
  private detectDocPaths(tree: string[]): string[] { ... }
}
```

   Key implementation:
   - `bootstrap()`: Calls `walkTree` to get the file tree (depth-limited, count-limited). Then runs `detectManifests`, `detectEntrypoints`, `detectTestPaths`, `detectDocPaths`. Assembles the `IndexPayload` object with repoInfo fields, `indexedAt: new Date().toISOString()`, and `truncated` flag if tree was capped.
   - `walkTree(basePath, relativePath, depth, results)`: Uses `readdir(join(basePath, relativePath), { withFileTypes: true })`. For each entry: if file, push `join(relativePath, entry.name)` to results. If directory and depth < MAX_DEPTH, recurse. Stop adding if `results.length >= MAX_TREE_ENTRIES`. Return `true` if truncated.
   - `detectManifests()`: For each manifest pattern file, check if it exists in tree. If found, read the file content with `readFile` (parse JSON for package.json; store raw for others). Return array of `ManifestEntry`.
   - `detectEntrypoints()`: Filter tree entries matching any ENTRYPOINT_PATTERNS (exact basename match).
   - `detectTestPaths()`: Filter tree entries matching TEST_PATTERNS regex.
   - `detectDocPaths()`: Filter tree for DOC_FILES + any paths starting with DOC_DIRS prefixes.
