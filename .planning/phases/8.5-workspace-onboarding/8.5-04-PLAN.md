    * Brownfield extraction context passed to InitRunner when onboarding
    * an existing repository. Mirrors ExtractionSnapshot from nest-core workspace module.
    * Defined here to avoid cross-package dependency (gsd-sdk must not import nest-core).
    */
   export interface BrownfieldContext {
     /** Git remote URL, branch info, dirty state */
     repoInfo: {
       remoteUrl: string | null;
       remoteName: string | null;
       baseBranch: string;
       currentBranch: string;
       isDirty: boolean;
       headSha: string;
     };
     /** Extracted repository index: tree, manifests, entrypoints, tests, docs */
     indexPayload: {
       tree: string[];
       manifests: Array<{ path: string; type: string; content: string }>;
       entrypoints: string[];
       testPaths: string[];
       docPaths: string[];
       remoteUrl: string | null;
       baseBranch: string;
       currentBranch: string;
       headSha: string;
       indexedAt: string;
       truncated?: boolean;
     };
     /** Absolute path to the workspace on the host */
     workspacePath: string;
     /** Project slug used for vault path */
     projectSlug: string;
   }
   ```

5. Add private method `buildBrownfieldInput(userInput: string): string`:
   ```typescript
   private buildBrownfieldInput(userInput: string): string {
     const bf = this.brownfield!;
     const sections = [
       userInput,
       '',
       '<extraction_snapshot>',
       '## Repository Info',
       `- Remote: ${bf.repoInfo.remoteUrl ?? 'local-only'}`,
       `- Base branch: ${bf.repoInfo.baseBranch}`,
       `- Current branch: ${bf.repoInfo.currentBranch}`,
       `- HEAD: ${bf.repoInfo.headSha}`,
       `- Dirty: ${bf.repoInfo.isDirty}`,
       '',
       '## File Tree (top 50 entries)',
       ...bf.indexPayload.tree.slice(0, 50).map(f => `- ${f}`),
       bf.indexPayload.truncated ? '... (truncated)' : '',
       '',
       '## Manifests',
       ...bf.indexPayload.manifests.map(m => [
         `### ${m.path} (${m.type})`,
         '```',
         m.content.slice(0, 500),
         '```',
       ].join('\n')),
       '',
       '## Entrypoints',
       ...bf.indexPayload.entrypoints.map(e => `- ${e}`),
       '',
       '## Test Paths',
       ...bf.indexPayload.testPaths.slice(0, 20).map(t => `- ${t}`),
       '',
       '## Doc Paths',
       ...bf.indexPayload.docPaths.map(d => `- ${d}`),
       '</extraction_snapshot>',
     ];
     return sections.join('\n');
   }
   ```
