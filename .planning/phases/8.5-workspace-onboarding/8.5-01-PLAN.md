```typescript
@Injectable()
export class WorkspaceOnboardingService {
  private readonly logger = new Logger(WorkspaceOnboardingService.name);

  constructor(
    @InjectRepository(WorkspaceSnapshot)
    private readonly snapshotRepo: Repository<WorkspaceSnapshot>,
    @InjectRepository(Project)
    private readonly projectRepo: Repository<Project>,
    private readonly ksm: KsmService,
  ) {}

  async initWorkspace(dto: InitWorkspaceDto): Promise<{ project: Project; snapshot: WorkspaceSnapshot }> { ... }
  async validateRepo(workspacePath: string): Promise<RepoInfo> { ... }
  async validateCredentials(projectId: string, workspacePath: string, repoInfo: RepoInfo, credential?: { type: 'ssh'; key: string } | { type: 'https'; pat: string }): Promise<void> { ... }
  async generateVaultDocs(projectSlug: string, indexPayload: IndexPayload): Promise<void> { ... }
  async persistSnapshot(projectId: string, workspacePath: string, repoInfo: RepoInfo, indexPayload: IndexPayload): Promise<WorkspaceSnapshot> { ... }
}
```

   Key implementation details:
   - `initWorkspace`: If `dto.repoUrl` is provided, clone it to `dto.workspacePath` using `simpleGit().clone(dto.repoUrl, dto.workspacePath)`. Then call `validateRepo`, create/find the Project entity, call `persistSnapshot`. Return `{ project, snapshot }`.
   - `validateRepo`: Use `simpleGit(workspacePath)` API: `checkIsRepo(CheckRepoActions.IS_REPO_ROOT)`, `getRemotes(true)`, `status()`, `revparse(['HEAD'])`. Detect base branch via `git.raw(['symbolic-ref', '--short', 'refs/remotes/origin/HEAD'])` with fallback to 'main'.
   - `validateCredentials`: If `credential.type === 'ssh'`, store via `this.ksm.createSecret('workspace-ssh-key-' + projectId, credential.key, 'project' as SecretScope, projectId)`. If `credential.type === 'https'`, store as `workspace-https-pat-<projectId>`. Then run `simpleGit(workspacePath).listRemote(['--heads', repoInfo.remoteUrl])` to test read access.
   - `generateVaultDocs`: Write 5 markdown files to `resolve(process.cwd(), 'volumes', 'obsidian', 'projects', projectSlug)`. Each file has YAML frontmatter and stub content derived from the `indexPayload` (manifests, entrypoints, tree). Use `mkdir` with `{ recursive: true }` for the directory.
   - `persistSnapshot`: Use `this.snapshotRepo.findOne({ where: { projectId } })` to check for existing. If found, update in place. If not, create new. Save via `this.snapshotRepo.save()`.
