   ```typescript
   import { Injectable, Logger } from '@nestjs/common';
   import simpleGit from 'simple-git';
   import { mkdir, rm } from 'node:fs/promises';
   import { join, resolve } from 'node:path';

   const RUNS_BASE = resolve(process.cwd(), 'volumes', 'runs');

   @Injectable()
   export class EphemeralWorkspaceService {
     private readonly logger = new Logger(EphemeralWorkspaceService.name);

     async cloneForRun(hostWorkspacePath: string, runId: string): Promise<string> {
       const runPath = join(RUNS_BASE, runId);
       await mkdir(RUNS_BASE, { recursive: true });

       const git = simpleGit();
       await git.clone(hostWorkspacePath, runPath, ['--local', '--no-hardlinks']);

       this.logger.log(`Ephemeral workspace cloned for run ${runId} at ${runPath}`);
       return runPath;
     }

     async cleanupRun(runId: string): Promise<void> {
       const runPath = join(RUNS_BASE, runId);
       await rm(runPath, { recursive: true, force: true });
       this.logger.log(`Ephemeral workspace cleaned for run ${runId}`);
     }
   }
   ```

3. Implement `promotion.service.ts`:
   ```typescript
   import { Injectable, Logger } from '@nestjs/common';
   import simpleGit from 'simple-git';
   import { join, resolve } from 'node:path';
   import { writeFile, mkdtemp, rm } from 'node:fs/promises';
   import { tmpdir } from 'node:os';

   const RUNS_BASE = resolve(process.cwd(), 'volumes', 'runs');

   @Injectable()
   export class PromotionService {
     private readonly logger = new Logger(PromotionService.name);

     /** Generate diff from ephemeral run workspace and apply it to host workspace */
     async promoteViaPatch(runId: string, hostPath: string): Promise<void> {
       const runPath = join(RUNS_BASE, runId);
       const runGit = simpleGit(runPath);

       // Generate diff of all changes in the run workspace
       const diff = await runGit.diff(['HEAD~1']);
       if (!diff.trim()) {
         this.logger.warn(`No changes to promote from run ${runId}`);
         return;
       }

       // Write diff to temp file and apply to host
       const tmpDir = await mkdtemp(join(tmpdir(), 'bridge-patch-'));
       const patchPath = join(tmpDir, 'changes.patch');
       await writeFile(patchPath, diff, 'utf-8');

       const hostGit = simpleGit(hostPath);
       await hostGit.raw(['apply', patchPath]);

       await rm(tmpDir, { recursive: true, force: true });
       this.logger.log(`Promoted run ${runId} to ${hostPath} via patch`);
     }

     /** Cherry-pick specific commits from ephemeral run workspace into host */
     async promoteViaCherryPick(runId: string, hostPath: string, commitShas: string[]): Promise<void> {
       const runPath = join(RUNS_BASE, runId);
       const hostGit = simpleGit(hostPath);

       // Add ephemeral workspace as temporary remote
       const remoteName = `run-${runId}`;
       await hostGit.addRemote(remoteName, runPath);
       await hostGit.fetch(remoteName);

       // Cherry-pick each commit
       for (const sha of commitShas) {
         await hostGit.raw(['cherry-pick', sha]);
       }

       // Remove temporary remote
       await hostGit.removeRemote(remoteName);
       this.logger.log(`Promoted ${commitShas.length} commits from run ${runId} to ${hostPath} via cherry-pick`);
     }

     /** Push the run branch directly to the remote */
     async promoteViaPush(runId: string, remoteName: string, branchName: string): Promise<void> {
       const runPath = join(RUNS_BASE, runId);
       const runGit = simpleGit(runPath);

       await runGit.push(remoteName, branchName);
       this.logger.log(`Pushed branch ${branchName} from run ${runId} to ${remoteName}`);
     }
   }
   ```
