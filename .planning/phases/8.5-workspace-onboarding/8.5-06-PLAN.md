1. Create `packages/cli/src/client.ts` — HTTP client for CLI-to-daemon communication:
   ```typescript
   export interface InitResponse {
     success: boolean;
     projectId?: string;
     slug?: string;
     snapshotId?: string;
     vaultDocs?: string[];
     error?: string;
   }

   export class DaemonClient {
     constructor(private readonly baseUrl: string) {}

     async initWorkspace(input: {
       workspacePath: string;
       repoUrl?: string;
       projectName: string;
       slug: string;
     }): Promise<InitResponse> {
       const res = await fetch(`${this.baseUrl}/internal/workspaces/init`, {
         method: 'POST',
         headers: { 'Content-Type': 'application/json' },
         body: JSON.stringify(input),
       });

       if (!res.ok) {
         const body = await res.text();
         throw new Error(`Daemon error (${res.status}): ${body}`);
       }

       return res.json() as Promise<InitResponse>;
     }

     async healthCheck(): Promise<boolean> {
       try {
         const res = await fetch(`${this.baseUrl}/health`, {
           signal: AbortSignal.timeout(3000),
         });
         return res.ok;
       } catch {
         return false;
       }
     }
   }
   ```

2. Create `packages/nest-core/src/module/workspace/workspace.controller.ts`:
   Per D-08, the controller must run BOTH stages in one command: extraction then AI doc generation via InitRunner brownfield mode. The controller:
   - Injects `WorkspaceOnboardingService` (Plan 03) for extraction
   - Injects `BrainService` (Plan 04 dependency — provides ProviderAdapter for InitRunner)
   - After `initOnboarding()` returns `ExtractionSnapshot`, converts it to `BrownfieldContext` and creates an `InitRunner` instance
   - Runs `InitRunner.run()` with the brownfield context to generate AI-powered docs

   ```typescript
   import { Body, Controller, Post, Logger, HttpCode } from '@nestjs/common';
   import { WorkspaceOnboardingService } from './workspace-onboarding.service';
   import { BrainService } from '../brain/brain.service';
   import { InitRunner } from '@bridge-ai/gsd-sdk';
   import type { BrownfieldContext } from '@bridge-ai/gsd-sdk';
   import type { OnboardingInput, ExtractionSnapshot } from './types';

   interface InitRequestBody {
     workspacePath: string;
     repoUrl?: string;
     projectName: string;
     slug: string;
   }

   interface InitResponseBody {
     success: boolean;
     projectId?: string;
     slug: string;
     snapshotId?: string;
     vaultDocs?: string[];
     error?: string;
   }

   @Controller('internal/workspaces')
   export class WorkspaceController {
     private readonly logger = new Logger(WorkspaceController.name);

     constructor(
       private readonly onboardingService: WorkspaceOnboardingService,
       private readonly brainService: BrainService,
     ) {}

     @Post('init')
     @HttpCode(200)
     async init(@Body() body: InitRequestBody): Promise<InitResponseBody> {
       this.logger.log(`Onboarding workspace: ${body.workspacePath} (slug: ${body.slug})`);

       try {
         const input: OnboardingInput = {
           workspacePath: body.workspacePath,
           repoUrl: body.repoUrl,
           projectName: body.projectName,
           slug: body.slug,
         };

         // Stage 1: Extraction (offline, no AI) — per D-08
         const snapshot: ExtractionSnapshot = await this.onboardingService.initOnboarding(input);
         this.logger.log(`Extraction complete for ${body.slug}, starting AI doc generation`);

         // Stage 2: AI doc generation via InitRunner brownfield — per D-08, D-11, D-12
         // Convert ExtractionSnapshot to BrownfieldContext (same shape, defined in gsd-sdk)
         const brownfield: BrownfieldContext = {
           repoInfo: snapshot.repoInfo,
           indexPayload: snapshot.indexPayload,
           workspacePath: snapshot.workspacePath,
           projectSlug: snapshot.projectSlug,
         };

         const vaultDocsDir = `projects/${body.slug}`;

         try {
           const runner = new InitRunner({
             projectDir: body.workspacePath,
             tools: null as any,       // Read from actual GSDTools factory — executor should check Plan 04 SUMMARY for correct wiring
             eventStream: null as any,  // Minimal no-op event stream for daemon-side execution
             adapter: this.brainService,
             brownfield,
           });

           const result = await runner.run(
             `Brownfield onboarding for project "${body.projectName}" at ${body.workspacePath}`,
           );

           this.logger.log(`InitRunner complete for ${body.slug}: ${result.steps?.length ?? 0} steps`);
         } catch (initRunnerErr) {
           // InitRunner failure is non-fatal: extraction succeeded, AI docs are best-effort
           // The extraction-based vault docs from initOnboarding() are already written
           this.logger.warn(
             `InitRunner brownfield failed (non-fatal): ${initRunnerErr instanceof Error ? initRunnerErr.message : initRunnerErr}. Extraction-based vault docs are still available.`,
           );
         }

         this.logger.log(`Onboarding complete for ${body.slug}`);

         return {
           success: true,
           slug: body.slug,
           vaultDocs: [
             `${vaultDocsDir}/project.md`,
             `${vaultDocsDir}/architecture.md`,
             `${vaultDocsDir}/stack.md`,
             `${vaultDocsDir}/decisions.md`,
             `${vaultDocsDir}/runbook.md`,
           ],
         };
       } catch (err) {
         this.logger.error(`Onboarding failed: ${err instanceof Error ? err.message : err}`);
         return {
           success: false,
           slug: body.slug,
           error: err instanceof Error ? err.message : String(err),
         };
       }
     }
   }
   ```

   IMPORTANT implementation notes for the executor:
   - The `tools` and `eventStream` constructor args need real values. Read `packages/gsd-sdk/src/init-runner.ts` to determine the minimal deps needed. If `GSDTools` requires filesystem access, create a minimal implementation or import the factory. If `GSDEventStream` can be a no-op logger, create one inline. The key contract is: `InitRunner` must receive `brownfield` and `adapter` (BrainService).
   - `BrainService` implements `ProviderAdapter` (verified in Phase 2) — it is the correct adapter to pass.
   - If InitRunner throws (e.g., no AI provider configured), the error is caught and logged as non-fatal. The extraction-based vault docs from `initOnboarding()` are still the baseline output. This is graceful degradation, not a TODO deferral.

4. Update `packages/cli/src/commands/init.ts` — replace the stub with real implementation:
   ```typescript
   import { resolve } from 'node:path';
   import { existsSync } from 'node:fs';
   import { ensureDaemon } from '../daemon.js';
   import { DaemonClient } from '../client.js';

   export interface InitOptions {
     workspace: string;
     repo?: string;
     name?: string;
     slug?: string;
   }

   export async function runInit(options: InitOptions): Promise<void> {
     const workspacePath = resolve(options.workspace);

     if (!existsSync(workspacePath)) {
       console.error(`[bridge] Error: workspace path does not exist: ${workspacePath}`);
       process.exit(1);
     }

     const projectName = options.name ?? workspacePath.split('/').pop() ?? 'unnamed';
     const projectSlug = options.slug ?? projectName.toLowerCase().replace(/[^a-z0-9-]/g, '-').replace(/-+/g, '-');

     console.log(`[bridge] Initializing: ${projectName} (${projectSlug})`);
     console.log(`[bridge] Workspace: ${workspacePath}`);

     // Stage 0: Ensure daemon is running
     let daemonUrl: string;
     try {
       daemonUrl = await ensureDaemon();
       console.log(`[bridge] Daemon connected at ${daemonUrl}`);
     } catch (err) {
       console.error(`[bridge] Failed to start daemon: ${err instanceof Error ? err.message : err}`);
       process.exit(1);
     }

     // Stage 1+2: Send to daemon for extraction + AI doc generation (per D-08)
     const client = new DaemonClient(daemonUrl);
     try {
       console.log('[bridge] Running extraction and AI doc generation...');
       const result = await client.initWorkspace({
         workspacePath,
         repoUrl: options.repo,
         projectName,
         slug: projectSlug,
       });

       if (!result.success) {
         console.error(`[bridge] Onboarding failed: ${result.error}`);
         process.exit(1);
       }

       console.log('[bridge] Onboarding complete!');
       console.log(`[bridge] Vault docs created:`);
       for (const doc of result.vaultDocs ?? []) {
         console.log(`  - ${doc}`);
       }
     } catch (err) {
       console.error(`[bridge] Error: ${err instanceof Error ? err.message : err}`);
       process.exit(1);
     }
   }
   ```
